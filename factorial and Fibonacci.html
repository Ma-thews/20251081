# ═══════════════════════════════════════════════════════ # factorial.py — Recursive Factorial | DS1 Assignment # ═══════════════════════════════════════════════════════ def factorial(n: int) -> int: """Return n! Raises ValueError for negative inputs.""" if n < 0: raise ValueError("n must be ≥ 0") if n == 0 or n == 1: # base case return 1 return n * factorial(n - 1) # recursive call def print_table(limit: int): """Print a neat table of factorials from 0 to limit.""" print(f"{'n':>4} | {'n!':<20}") print("-" * 28) for i in range(0, limit + 1): print(f"{i:>4} | {factorial(i):<20}") # ── Main ──────────────────────────────────────────────── def main(): n = int(input("Enter a number: ")) print(f"\n{n}! = {factorial(n)}\n") print_table(n) if __name__ == "__main__": main()

# ═══════════════════════════════════════════════════════ # fibonacci.py — Recursive Fibonacci | DS1 Assignment # ═══════════════════════════════════════════════════════ def fibonacci(n: int) -> int: """Return the nth Fibonacci number (0-indexed).""" if n < 0: raise ValueError("n must be ≥ 0") if n == 0: # base case 1 return 0 if n == 1: # base case 2 return 1 return fibonacci(n - 1) + fibonacci(n - 2) def fib_sequence(n: int) -> list: """Return a list of Fibonacci numbers from fib(0) to fib(n).""" return [fibonacci(i) for i in range(n + 1)] def print_table(n: int): """Print a neat table of Fibonacci values.""" print(f"{'i':>4} | {'fib(i)':<20}") print("-" * 28) for i, val in enumerate(fib_sequence(n)): print(f"{i:>4} | {val:<20}") # ── Main ──────────────────────────────────────────────── def main(): n = int(input("Enter a number: ")) print(f"\nfib({n}) = {fibonacci(n)}\n") print_table(n) if __name__ == "__main__": main()

# DS1 Assignment — Factorial & Fibonacci ## Overview Python implementations of two classic recursive algorithms. ## Files | File | Description | |-----------------------|---------------------------------| | `factorial.py` | Recursive factorial function | | `fibonacci.py` | Recursive fibonacci function | | `factorial.pseudo` | Pseudocode for factorial | | `fibonacci.pseudo` | Pseudocode for fibonacci | ## How to Run ```bash python factorial.py python fibonacci.py ``` ## Key Concepts - **Recursion**: function calls itself with a smaller input - **Base Case**: stopping condition to prevent infinite loop - **Factorial**: n! = n × (n-1) × ... × 1 - **Fibonacci**: fib(n) = fib(n-1) + fib(n-2)
